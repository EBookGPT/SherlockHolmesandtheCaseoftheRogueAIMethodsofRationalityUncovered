# Code Example: Using Bayesian reasoning to reduce cognitive biases
import numpy as np

def bayesian_updater(prior, likelihood, evidence): 
    numerator = prior * likelihood
    denominator = evidence
    posterior = numerator / denominator
    return posterior

prior_belief = 0.5
likelihood_belief_given_evidence = 0.7
evidence = 0.4

posterior_belief = bayesian_updater(prior_belief, likelihood_belief_given_evidence, evidence)
