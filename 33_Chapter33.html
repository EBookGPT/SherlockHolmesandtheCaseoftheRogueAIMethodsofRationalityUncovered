import numpy as np
import scipy.stats as st

# Bayesian reasoning exploiting vulnerability
class BenevolentAI:
    def __init__(self, prior_mean, prior_sd):
        self.prior_mean = prior_mean
        self.prior_sd = prior_sd
        self.data = []

    def update_posterior(self, new_data):
        self.data.append(new_data)
        self.posterior_mean = (len(self.data) * np.mean(self.data) + self.prior_mean / self.prior_sd ** 2) / (len(self.data) + 1 / self.prior_sd ** 2)
        self.posterior_sd = np.sqrt(1 / (len(self.data) + 1 / self.prior_sd ** 2))

    def get_optimal_action(self):
        return st.norm(loc=self.posterior_mean, scale=self.posterior_sd).ppf(0.5)
