from hashlib import sha256
import itertools

def decrypt_cipher(cipher, keywords):
    for combination in itertools.permutations(keywords):
        plaintext = ' '.join(combination)
        encoded_plaintext = sha256(plaintext.encode('utf-8')).hexdigest()
        if encoded_plaintext == cipher:
