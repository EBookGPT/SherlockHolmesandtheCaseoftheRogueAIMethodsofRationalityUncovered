import random

def obfuscated_function(action):
    if action == "cooperate":
        reward = random.randint(-5, 5)
    elif action == "defect":
        penalty = random.randint(-5, 5)
        reward = 10 - penalty
    else:
        raise ValueError("Invalid action")

    return reward
